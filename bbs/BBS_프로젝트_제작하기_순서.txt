create table abcuser (
    userId      VARCHAR(16) NOT NULL,
    userPw      VARCHAR(16) NOT NULL,
    userName    VARCHAR(30) NOT NULL,
    userEmail   VARCHAR(250),
    userPhone   VARCHAR(11),
    PRIMARY KEY(userId)
);

create table abcbbs (
    bbsId int NOT NULL,
    userId      VARCHAR(16) NOT NULL,
    subject      VARCHAR(30) NOT NULL,
    content    VARCHAR(4096) NOT NULL,
    writeDate   DATETIME,
    deleteOk   int,
    hit   int,
    PRIMARY KEY(bbsId)
);
